### 噫！汝之“狗”虽能叫，然若父类有protected成员，汝能访之乎？

```cpp
#include <iostream>
using namespace std;

class 动物 {
public:
    virtual void 叫() {  // 虚函数
        cout << "动物叫..." << endl;
    }
};

class 狗 : public 动物 {  // 正确：public 继承
public:
    void 叫() override {  // 重写虚函数
        cout << "汪汪汪！" << endl;
    }
};

int main() {
    动物* 动物指针 = new 狗();
    动物指针->叫();  // 多态调用
    delete 动物指针;
    return 0;
}

```
