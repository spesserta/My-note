
# 一、核心概念
## 1. 封装的本质
>将数据（成员变量） 和操作数据的方法（成员函数） 绑定在一起，同时隐藏内部实现细节，只对外暴露必要的接口，实现 "高内聚、低耦合"。<br>
>访问控制：通过访问限定符控制类成员的可见性，是封装的核心手段。<br>
>好处：
>- 保护数据安全，避免外部随意修改；
>- 降低代码耦合度，修改内部实现不影响外部使用；
>- 代码复用性更高。

# 二、类的基本定义与使用
## 1. 类的语法结构
```cpp

#include <iostream>
#include <string>
using namespace std;

// 类的定义：class关键字 + 类名 + 类体
class Person {
    // 私有成员（默认private，可省略）
    private:
        // 成员变量（属性）：建议加前缀区分（如m_），避免与参数名冲突
        string m_name;
        int m_age;

    // 公有成员：对外暴露的接口
    public:
        // 成员函数（方法）：设置/获取私有成员
        void setName(string name) {
            m_name = name;
        }

        void setAge(int age) {
            // 数据校验：封装的体现（外部无法直接修改，需通过校验）
            if (age < 0 || age > 150) {
                cout << "年龄非法！" << endl;
                m_age = 0;
                return;
            }
            m_age = age;
        }

        void showInfo() {
            cout << "姓名：" << m_name << "，年龄：" << m_age << endl;
        }
};

// 类的使用
int main() {
    // 1. 创建对象（实例化）
    Person p1;

    // 2. 调用公有成员函数（接口）
    p1.setName("张三");
    p1.setAge(20);
    p1.showInfo(); // 输出：姓名：张三，年龄：20

    // 3. 非法操作：类外无法访问private成员
    // p1.m_name = "李四"; // 编译报错
    // p1.m_age = -5;      // 编译报错

    // 4. 测试数据校验
    Person p2;
    p2.setName("李四");
    p2.setAge(200); // 输出：年龄非法！
    p2.showInfo();  // 输出：姓名：李四，年龄：0

    return 0;
}
```
## 2. 类的分文件编写（工程化规范）

>步骤 1：头文件（Person.h）

# 三、构造函数与析构函数
## 1. 构造函数（初始化对象）
>作用：创建对象时自动调用，初始化成员变量。<br>
>特性：
>- 函数名与类名完全相同，无返回值（连void都没有）；<br>
>- 可重载（多个构造函数，参数列表不同）；<br>
>- 若未自定义，编译器会生成默认构造函数（空实现）。<br>
## 2. 析构函数（清理对象）
> 作用：对象销毁时自动调用，释放资源（如堆内存、文件句柄）。<br>
> 特性：
> 函数名：~类名，无返回值、无参数；<br>
> 不可重载（只能有一个）；<br>
> 若未自定义，编译器生成默认析构函数（空实现）。<br>
## 3. 完整示例
```cpp

#include <iostream>
#include <string>
using namespace std;

class Student {
private:
    string m_name;
    int m_id;
    int* m_score; // 堆内存指针（演示析构函数清理资源）

public:
    // 1. 默认构造函数（无参）
    Student() {
        cout << "默认构造函数调用" << endl;
        m_name = "未知";
        m_id = 0;
        m_score = new int(0); // 堆内存分配
    }

    // 2. 有参构造函数（重载）
    Student(string name, int id, int score) {
        cout << "有参构造函数调用" << endl;
        m_name = name;
        m_id = id;
        m_score = new int(score); // 堆内存分配
    }

    // 3. 拷贝构造函数（深拷贝，避免浅拷贝问题）
    Student(const Student& other) {
        cout << "拷贝构造函数调用" << endl;
        m_name = other.m_name;
        m_id = other.m_id;
        // 深拷贝：重新分配内存，复制值
        m_score = new int(*other.m_score);
    }

    // 4. 析构函数：释放堆内存
    ~Student() {
        cout << "析构函数调用：释放" << m_name << "的资源" << endl;
        if (m_score != nullptr) {
            delete m_score; // 释放堆内存
            m_score = nullptr; // 防止野指针
        }
    }

    // 成员函数
    void showInfo() {
        cout << "姓名：" << m_name << "，学号：" << m_id 
             << "，成绩：" << *m_score << endl;
    }
};

int main() {
    // 调用默认构造
    Student s1;
    s1.showInfo(); // 姓名：未知，学号：0，成绩：0

    // 调用有参构造
    Student s2("小明", 1001, 95);
    s2.showInfo(); // 姓名：小明，学号：1001，成绩：95

    // 调用拷贝构造
    Student s3 = s2;
    s3.showInfo(); // 姓名：小明，学号：1001，成绩：95

    return 0;
}
```

# 四、封装的高级用法
## 1. 友元（打破封装的特例）
>友元允许外部函数 / 类访问类的私有成员（谨慎使用，避免破坏封装）：
```cpp

#include <iostream>
#include <string>
using namespace std;

class Car {
    // 声明友元函数：允许该函数访问私有成员
    friend void showCarInfo(Car& car);
    // 声明友元类：允许整个类访问私有成员
    friend class Garage;

private:
    string m_brand;
    int m_price;

public:
    Car(string brand, int price) {
        m_brand = brand;
        m_price = price;
    }
};

// 友元函数实现
void showCarInfo(Car& car) {
    // 直接访问私有成员
    cout << "品牌：" << car.m_brand << "，价格：" << car.m_price << endl;
}

// 友元类定义
class Garage {
public:
    void repairCar(Car& car) {
        // 直接修改私有成员
        car.m_price -= 1000;
        cout << "维修后价格：" << car.m_price << endl;
    }
};

int main() {
    Car c("特斯拉", 300000);
    showCarInfo(c); // 品牌：特斯拉，价格：300000

    Garage g;
    g.repairCar(c); // 维修后价格：299000
    showCarInfo(c); // 品牌：特斯拉，价格：299000

    return 0;
}
```
## 2. 常成员（const）
### （1）常成员变量：必须在构造函数初始化列表初始化
```cpp

class Circle {
private:
    const double m_pi; // 常成员变量（不可修改）
    double m_radius;

public:
    // 初始化列表：初始化常成员变量
    Circle(double radius) : m_pi(3.14159), m_radius(radius) {}

    double getArea() {
        return m_pi * m_radius * m_radius;
    }
};
```
### （2）常成员函数：不能修改成员变量
```cpp

class Book {
private:
    string m_title;
    int m_pages;

public:
    Book(string title, int pages) : m_title(title), m_pages(pages) {}

    // 常成员函数：末尾加const
    string getTitle() const {
        // m_pages = 100; // 编译报错：常函数不能修改成员变量
        return m_title;
    }
};
```
# 五、封装的设计原则
>- 数据隐藏：成员变量私有化（private），仅通过公有接口（public成员函数）访问；
>- 接口简洁：对外暴露必要的接口，隐藏内部实现细节；
>-数据校验：在接口中对输入数据进行合法性校验；
>-资源管理：通过构造 / 析构函数统一管理资源（如堆内存、文件）；
>-慎用友元：友元会破坏封装，仅在必要时使用。
