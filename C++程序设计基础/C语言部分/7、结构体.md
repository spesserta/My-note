# 一、结构体的声明

>如下图就是一个标准的声明形式：其中person表示结构体的数据类型，timmy是声明结构体时顺便创建的一个结构体变量。结构体内部花括号里面的变量表示结构体的成员。下面的struct person是C语言中在其他区域创建结构体时固定加上的数据类型，struct表示这是一个结构体，person表示数据类型是person，剩下的表示变量名。

```c
#include<stdio.h>
struct person {     //表示人信息的结构体
	char name[20];  //名字
	int gender;     //性别
	double weigh;   //体重
	double heigh;   //身高
}timmy; //定义结构体的时候顺便定义一个timmy的人
int main() {
	struct person Jack;   //根据person结构体定义一个叫Jack的人
	struct person Mary; 
	retrun 0;
}
```


>可以通过打点的方式访问或者修改或者赋值结构体变量的成员。

```c
#include<stdio.h>
struct person {     //表示人信息的结构体
	char name[20];  //名字
	int gender;     //性别
	double weigh;   //体重
	double heigh;   //身高
}timmy; //定义结构体的时候顺便定义一个timmy的人
int main() {
	//打点给timmy信息赋值
	timmy.name = "timmy";
	timmy.gender = 1;
	timmy.weigh = 110.12;
	timmy.heigh = 170.22;
	//输出timmy的信息
	printf(timmy.name);
	printf(" %d %f %f",timmy.gender,timmy.heigh,timmy.weigh);
	retrun 0;
}
```

# 二、初始化结构体
> 结构体的初始化和数组的初始化类似：<br>

> 结构的初始化列表的写法需要注意如下4点：
>- 初始化列表由花括号包括。
>- 花括号内为结构成员需要被初始化的值。
>- 初始化值按照结构成员声明时的顺序依次排列
>- 每个初始化值之间由逗号分隔。

```c
#include<stdio.h>
struct person {     //表示人信息的结构体
	char name[20];  //名字
	int gender;     //性别
	double weigh;   //体重
	double heigh;   //身高
}; 
int main() {
	struct person Jack = {"Jack",1,123.22,180.12};  //初始化结构体
	
	retrun 0;
}
```

# 三、结构体数组

>和普通数组一样的操作

```c
#include<stdio.h>
struct person {     //表示人信息的结构体
	char name[20];  //名字
	int gender;     //性别
	double weigh;   //体重
	double heigh;   //身高
}; 
int main() {
	struct person people[3] = {   //申明结构体数组并且初始化
		{"Timmy",1,170,60 },
	    {"Jack" ,0,121,129},
		{"Mary" ,1,122,123},
	};
	 
	for (int i = 0;i < 3;i++) {     //输出结构体数组需要申明一个中间变量
		struct person per = people[i];
		printf("%s",   per.name);
		printf("%d",   per.gender);
		printf("%.2f", per.weigh);
		printf("%.2f", per.heigh);
	}

	retrun 0;
}
```

# 四、嵌套结构体

>一个结构体作为单独变量在另外一个结构体的声明内就是嵌套结构体。初始化的时候内部对应位置多加一个花括号即可。>访问的时候也相应的多加一个点号即可。

```c
#include<stdio.h>
struct privacyi {  //表示人隐私信息的结构体
	char phone[20];
	char email[20];
	unsigned int Id;
};
struct person {     //表示人基础信息的结构体
	char name[20];  //名字
	int gender;     //性别
	double weigh;   //体重
	double heigh;   //身高
	struct privacyi p;
}; 

int main() {
	
	struct person timmy = {
		"timmy", 1, 170.22 , 60.00,
		{"18270729122", "28728121@qq.com", "2182761882121"}
	};

	//输出隐私信息
	printf(timmy.p.email);
	printf("\n");
	printf(timmy.p.phone);
	printf("\n");
	retrun 0;
}
```

# 六、结构体在函数中传递


>结构体的变量类型可以作为形参实参和返回值进行传递，在变量类型前面加上struct即可让C语言知道这是一个结构体。


```c
#include<stdio.h>
struct privacyi {  //表示人隐私信息的结构体
	char phone[20];
	char email[20];
	unsigned int Id;
};
struct person {     //表示人基础信息的结构体
	char name[20];  //名字
	int gender;     //性别
	double weigh;   //体重
	double heigh;   //身高
	struct privacyi p;
}; 

struct person defaultInit() {  //返回值为结构体类型
	struct person per;
	strcpy(per.name,"default"); //给结构体变量赋初始值
	per.gender = 1;
	per.heigh = 170.21;
	per.weigh = 212;
	return per;
}

int main() {
	
	struct person Jimmy;
	Jimmy = defaultInit();  //返回值是结构体类型，接收返回值的变量也应该是结构体类型

	retrun 0;
}
```

# 七、共用体union
## 1、结构体和共用体的内存排列

>根据实践可知结构体的内存排布情况如下：<br>
>结构体所占空间的大小等于或者大于各数据成员的内存大小之和。



>可以看出char和short之间有1个空格，short和long long 之间有4个空格字节，这种情况就是内存对齐，虽然会浪费一些空间，但是有助于结构体更快的访问数据成员。<br>
>以下是共用体的排列规则：




可以看出3个数据类型共享一个内存空间，共用体的大小取决于数据类型最大的变量所占用的内存空间，这就意味着存储一个成员后，将覆盖掉其他成员的数据。
共用体也有它的用处，假设要用结构体存储一个数据，但是这个数据可能是char型可能是int或者其他类型，这时候就可以用上共用体来存储这个不确定类型的数据。
