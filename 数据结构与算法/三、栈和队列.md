# 一、栈的基本概念

## 1、定义：栈是只允许在一端进行插入或者删除操作的线性表。

>-栈顶：允许插入和删除的一端<br>
>-栈底：不允许插入和删除的一端<br>
>-空栈：没有元素的栈<br>

## 2、特点：后进先出，先进后出

## 3、基本操作：
>-创建（初始化栈）<br>
>-销毁（销毁栈）<br>
>-增（进栈）<br>
>-删（出栈）<br>
>-查（读取栈元素）<br>
>-判断栈空

# 二、栈的顺序存储实现

## 1、栈的结构体表示：
```cpp
    #define MaxSize 10                  //栈中元素最大个数
    typedef struct{
         ElemType data[MaxSize]；   //data域
         int top;                                //栈顶指针（位置，不是真指针）
    }SqStack；
```

## 2、初始化顺序栈：
```cpp
   void InitStack(SqStack &S){
     S.top=-1;                                //假设最上面元素为栈顶位置，则初始化栈顶指针为-1
    }
```
## 3、判断栈空：
```cpp
   bool StackEmpty(SqStack S){
       if(S.top=-1)   return true;
       else return false;
   }
```

## 4、新元素入栈：
```cpp
  bool Push(SqStack &S,ElemType x){
      if(S.top==MaxSize-1)  return false;   //栈满就不入栈
      S.top++;                                            //修改栈顶指针
      S.data[S.top]=x;                                //新元素入栈到栈顶指针处
      return true;
   }
```
## 5、出栈：
```cpp
   bool Pop(SqStack &S,ElemType &x){      //x拿来备份数据的
      if(S.top==-1)   return false;                   //栈空还出什么栈
      x=S.data[S.top];                                     //备份出栈数据
      S.top--;                                                  //修改栈顶指针
      return true;
     }
```
## 6、读取栈顶元素
```cpp
     bool GetTop(SqStack S,ElemType &x){
	      if(S.top==-1)   return false;                         //空栈
            x=S.data[S.top];        
		 return true;        
     }
```
## 7、共享栈：两个栈共享一片空间
```cpp
    #define MaxSize 10
    typedef struct{                            //结构体定义
       ElemType data[MaxSize];         //data域
       int top0；
       int top1；
    }ShStack；

   void InitStack(ShStack &S)           //初始化
     {     
        S.top0=-1；
        S.top1=MaxSize；
    }

    bool  OccupyStack(ShStack &S) //判断栈满
    {
		if(top0+1==top1)   return true;
		else return false;
    }
   
    bool EmptyStack(ShStack &S)     //判断栈空
    {
	        If(( S.top0=-1)&&(S.top1=MaxSize))     return true;
		else return false;
     }
```

# 五、队列的顺序存储实现

## 1、结构体
```cpp
 #define MaxSize 10
 typedef struct{
     ElemType data[MaxSize];
     int front,rear;
  }SqQueue;
```
## 2、入队
```cpp
  bool EnQueue(SqQueue &Q,ElemType x){
     if((Q.rear+1)%MaxSize==Q.front)   return false;   //队满
     Q.data[Q.rear]=x;                                            //逻辑上入队
     Q.rear=(Q.rear+1)%MaxSize;                          //队尾逻辑上+1
     return true;
  }
```
## 3、出队
```cpp
   bool DeQueue(SqQueue &Q,ElemType &x){
      if(Q.rear==Q.front)   return false;   //队空
       x=Q.data[Q.front+1];                     //逻辑上出队
      Q.front=(Q.front+1)%MaxSize;      //队头逻辑上-1
       return true;
  }
```
## 4、获得队头元素的值，用x返回
```cpp
   bool GetHead(SqQueue Q,ElemType &x){
       if(Q.rear==Q.front)   return false;
        x=Q.data[Q.front];
     return true;
  }
```
### 5、判断队空 :   Q.rear==Q.front

### 6、判断队满  :  (Q.rear+1)%MaxSize==Q.front

### 7、判断元素个数：  (rear+MaxSize-front)%MaxSize

### 8、区分队满和队空

>以上区分队满和队空的做法会牺牲一个存储空间，为了利用那最后一点空间，可以在结构体内设立一个tag，插入时tag=1，删除时候tag=0，当front==rear&&tag==1时队满，当front==rear&&tag==0时队空。




# 六、队列的链式存储实现

## 1、结构体
```cpp
  typedef struct LinkNode{
      ElemType data;
      struct LinkNode *next; 
  }LinkNode;

  typedef struct{
     LinkNode *front,*rear;                                                  //队头指针和队尾指针
  }LinkQueue;
```
## 2、初始化:
```cpp
  void InitQueue(LinkQueue &Q){                                  //带头节点
     Q.front=Q.rear=(LinkNode*)malloc(sizeof(LinkNode));
     Q.front->next=NULL;                                                //队头用于出队，因此初始值是NULL
  }

    void InitQueue(LinkQueue &Q){                             //不带头结点
	Q.front=NULL;
	Q.rear=NULL;
  }
```
## 3、判断队列是否为空:
```cpp
  bool IsEmpty(LinkQueue Q){                                        //带头节点
      if(Q.front==Q.rear)  return true;
      else return false;
  }

    bool IsEmpty(LinkQueue Q){                                       //不带头结点
        if(Q.front==NULL)  return true;
        else return false;
}

```
## 4、入队
```cpp
   void EnQueue(LinkQueue &Q,ElemType x){       //带头节点
         LinkNode *s=(LinkNode*)malloc(sizeof(LinkNode));
         s->data=x;
         s->next=NULL;
         Q.rear->next=s;
            Q.rear=s;
   }

   void EnQueue(LinkQueue &Q,ElemType x){           //不带头结点
     LinkNode *s=(LinkNode*)malloc(sizeof(LinkNode));
     s->data=x;
     s->next=NULL;
     if(Q.front==NULL){                                              //没有元素时rear和front初始值为NULL
           Q.front=s;                                            //此刻s是第一个节点
           Q.rear=s;
     }
    else{
	Q.rear->next=s;
	Q.rear=s;
     }
 }
```
## 5、出队
```cpp
bool DeQueue(LinkQueue &Q,ElemType &x){       //带头结点
    if(Q.front==Q.rear)    return false;                      //队空
    LinkNode *p=Q.front->next;                              //Q.front为头结点，p指向要删除元素
    x=p->data;                   
    Q.front->next=p->next ;                                    //删除
     if(Q.rear==p)                                                     //删除的是最后一个节点
         Q.rear=Q.front;                                             //改队尾指针
    free(p);                                                               //释放
    return true;
}

bool DeQueue(LinkQueue &Q,ElemType &x){     //带头结点
         if(Q.front==NULL)  return false;
         LinkNode *p=Q.front;                                 //Q.front为第一个节点，删的就是它
         x=p->data;
         Q.front=p->next;                                        //删除
         if(Q.rear==p) {                                            //如果删除的是最后一个元素
                  Q.front=NULL; 
                  Q.rear=NULL;
            }
         free(p);
         return true;
}

```

# 八、括号匹配问题：

### 1、用栈实现括号匹配

>就依次扫描所有字符，遇到左括号就入栈，遇到右括号就弹出栈顶元素检查是否一样。当然也有匹配失败的情况： 左括号匹配失败（匹配完栈里面还有左括号）、右括号匹配失败（有右括号但是栈空没左括号了）、左右括号不匹配（左括号不等于右括号）

## 2、括号匹配流程图：

<img width="942" height="586" alt="image" src="https://github.com/user-attachments/assets/dee49562-0e58-4874-8ce8-c76f69b6b7d4" />



## 3、算法实现
```cpp
bool bracketCheck(char stl[],int length){
    SqStack S;
    InitStack(S);
    for(int i=0;i<length;i++)                                  //依次扫描所有字符
         if(str[i]=='('||str[i]=='['||str[i]=='{'){             //是左括号就压入栈中
             Push(S,str[i]);         
      }
    else{
         if(StackEmpty(S))   return false;                    //是右括号但是栈中没有元素
          char topElem;         //接收出栈的左括号
          Pop(S,topElem);   //左括号出栈
          if(str[i]==')'&&topElem!='(')  return false;    //左右不匹配就false
          if(str[i]==']'&&topElem!='[')  return false;
          if(str[i]=='}'&&topElem!='{')  return false;
        }
      return StackEmpty(S);                                       //匹配完但是栈中还有元素就false
} 
```
