### 一、为什么分层
>分层可以将把数据 → 业务 → 界面 分开，不乱、好维护、好扩展。 <br>
>就像家里： <br>
>厨房（存食材）= 数据库层 <br>
>客厅（处理食材、做饭）= 业务逻辑层 <br>
>餐厅（端给你吃）= 界面层 <br>
>如果混在一起，厨房、客厅、餐厅都在一个房间，那肯定乱死。


### 二、经典三层架构

#### 1、数据访问层 DAL（Data Access Layer）

##### 该层和数据库打交道

>- 打开连接
>- 执行 SQL
>- DataSet、DataAdapter、增删改查
>- 别的啥都不干

#### 2、业务逻辑层 BLL（Business Logic Layer）

##### 该层用来处理规则

>- 验证数据是否合法
>- 年龄不能为负
>- 用户名不能为空
>- 不能重复插入
>- 调用 DAL 层

#### 3、表示层 UI（User Interface）

##### 这层用来显示、接收输入

>- 控制台、WinForm、WPF、网页
>- 点击按钮
>- 显示数据
>- 调用 BLL 层


### 三、三层之间的关系（自上而下调用）

```
UI 界面层
   ↓（调用）
BLL 业务逻辑层
   ↓（调用）
DAL 数据访问层
   ↓（操作）
数据库
```
>注意这三层绝对不能跨层，例如UI不能直接访问数据库

```
你的项目
├── UI层（Program.cs 控制台界面）
├── BLL层（TestManager.cs 处理业务）
├── DAL层（TestService.cs 操作数据库）
└── Models层（Test.cs 实体类）
```

```
UI 只管显示
BLL 只管规则
DAL 只管数据库
Model 代表表结构
```

##### 1、Models层（和数据表的列名对应）实体类 User.cs

##### 2、DAL 层：数据访问 UserDAL.cs

```c#
using System;
using System.Data;
using System.Data.SqlClient;

public class UserDAL
{
    // 你的连接字符串
    private readonly string connStr = 
        "Data Source=DESKTOP-VL906FO\\SQLEXPRESS;Initial Catalog=awa;Integrated Security=True;";

    // 1. 查询所有用户（返回DataTable）
    public DataTable GetAllUsers()
    {
        string sql = "SELECT * FROM test";
        using (SqlDataAdapter adapter = new SqlDataAdapter(sql, connStr))
        {
            adapter.MissingSchemaAction = MissingSchemaAction.AddWithKey;
            DataTable dt = new DataTable();
            adapter.Fill(dt);
            return dt;
        }
    }

    // 2. 根据ID查询单个用户
    public User GetUserById(int id)
    {
        string sql = "SELECT * FROM test WHERE ID = @ID";
        using (SqlConnection conn = new SqlConnection(connStr))
        using (SqlCommand cmd = new SqlCommand(sql, conn))
        {
            cmd.Parameters.AddWithValue("@ID", id);
            conn.Open();
            using (SqlDataReader reader = cmd.ExecuteReader())
            {
                if (reader.Read())
                {
                    return new User
                    {
                        ID = (int)reader["ID"],
                        UserName = reader["UserName"].ToString(),
                        PassWord = reader["PassWord"].ToString(),
                        NickName = reader["NickName"].ToString(),
                        Sex = (int)reader["Sex"]
                    };
                }
                return null;
            }
        }
    }

    // 3. 新增用户
    public int AddUser(User user)
    {
        string sql = @"INSERT INTO test(UserName, PassWord, NickName, Sex)
                       VALUES(@UserName, @PassWord, @NickName, @Sex)";
        using (SqlConnection conn = new SqlConnection(connStr))
        using (SqlCommand cmd = new SqlCommand(sql, conn))
        {
            cmd.Parameters.AddWithValue("@UserName", user.UserName);
            cmd.Parameters.AddWithValue("@PassWord", user.PassWord);
            cmd.Parameters.AddWithValue("@NickName", user.NickName);
            cmd.Parameters.AddWithValue("@Sex", user.Sex);

            conn.Open();
            return cmd.ExecuteNonQuery(); // 返回受影响行数
        }
    }

    // 4. 修改用户（比如修改Sex和NickName）
    public int UpdateUser(User user)
    {
        string sql = @"UPDATE test 
                       SET NickName = @NickName, Sex = @Sex 
                       WHERE ID = @ID";
        using (SqlConnection conn = new SqlConnection(connStr))
        using (SqlCommand cmd = new SqlCommand(sql, conn))
        {
            cmd.Parameters.AddWithValue("@NickName", user.NickName);
            cmd.Parameters.AddWithValue("@Sex", user.Sex);
            cmd.Parameters.AddWithValue("@ID", user.ID);

            conn.Open();
            return cmd.ExecuteNonQuery();
        }
    }

    // 5. 删除用户
    public int DeleteUser(int id)
    {
        string sql = "DELETE FROM test WHERE ID = @ID";
        using (SqlConnection conn = new SqlConnection(connStr))
        using (SqlCommand cmd = new SqlCommand(sql, conn))
        {
            cmd.Parameters.AddWithValue("@ID", id);
            conn.Open();
            return cmd.ExecuteNonQuery();
        }
    }
}
```

##### 3、BLL 层：业务逻辑 UserBLL.cs

```c#
public class UserBLL
{
    // 实例化DAL对象
    private readonly UserDAL dal = new UserDAL();

    // 查询所有用户
    public DataTable GetAllUsers()
    {
        return dal.GetAllUsers();
    }

    // 查询单个用户
    public User GetUserById(int id)
    {
        if (id <= 0) return null; // 校验ID合法性
        return dal.GetUserById(id);
    }

    // 新增用户（带校验）
    public bool AddUser(User user)
    {
        // 业务校验
        if (string.IsNullOrWhiteSpace(user.UserName) || 
            string.IsNullOrWhiteSpace(user.PassWord) || 
            string.IsNullOrWhiteSpace(user.NickName))
        {
            Console.WriteLine("错误：用户名、密码、昵称不能为空！");
            return false;
        }
        if (user.Sex < 0 || user.Sex > 2) // 假设Sex只能是0/1/2
        {
            Console.WriteLine("错误：Sex只能是0/1/2！");
            return false;
        }

        int rows = dal.AddUser(user);
        return rows > 0;
    }

    // 修改用户（带校验）
    public bool UpdateUser(User user)
    {
        if (user.ID <= 0) return false;
        if (string.IsNullOrWhiteSpace(user.NickName))
        {
            Console.WriteLine("错误：昵称不能为空！");
            return false;
        }
        if (user.Sex < 0 || user.Sex > 2)
        {
            Console.WriteLine("错误：Sex只能是0/1/2！");
            return false;
        }

        int rows = dal.UpdateUser(user);
        return rows > 0;
    }

    // 删除用户（带校验）
    public bool DeleteUser(int id)
    {
        if (id <= 0) return false;

        int rows = dal.DeleteUser(id);
        return rows > 0;
    }
}
```



##### 4、UI 层：控制台交互 Program.cs

```c#
using System;
using System.Data;

class Program
{
    static void Main(string[] args)
    {
        UserBLL bll = new UserBLL();

        while (true)
        {
            Console.WriteLine("\n==== 用户管理系统 ====");
            Console.WriteLine("1. 查询所有用户");
            Console.WriteLine("2. 查询单个用户");
            Console.WriteLine("3. 新增用户");
            Console.WriteLine("4. 修改用户");
            Console.WriteLine("5. 删除用户");
            Console.WriteLine("0. 退出");
            Console.Write("请输入选择：");

            string choice = Console.ReadLine();
            switch (choice)
            {
                case "1":
                    ShowAllUsers(bll);
                    break;
                case "2":
                    ShowSingleUser(bll);
                    break;
                case "3":
                    AddNewUser(bll);
                    break;
                case "4":
                    UpdateUserInfo(bll);
                    break;
                case "5":
                    DeleteUserInfo(bll);
                    break;
                case "0":
                    return;
                default:
                    Console.WriteLine("无效选择！");
                    break;
            }
        }
    }

    // 显示所有用户
    static void ShowAllUsers(UserBLL bll)
    {
        DataTable dt = bll.GetAllUsers();
        Console.WriteLine("\nID\tUserName\tPassWord\tNickName\tSex");
        foreach (DataRow row in dt.Rows)
        {
            Console.WriteLine($"{row["ID"]}\t{row["UserName"]}\t\t{row["PassWord"]}\t\t{row["NickName"]}\t\t{row["Sex"]}");
        }
    }

    // 查询单个用户
    static void ShowSingleUser(UserBLL bll)
    {
        Console.Write("请输入用户ID：");
        int id = int.Parse(Console.ReadLine());
        User user = bll.GetUserById(id);
        if (user != null)
        {
            Console.WriteLine($"ID: {user.ID}, 用户名: {user.UserName}, 昵称: {user.NickName}, 性别: {user.Sex}");
        }
        else
        {
            Console.WriteLine("用户不存在！");
        }
    }

    // 新增用户
    static void AddNewUser(UserBLL bll)
    {
        Console.Write("请输入用户名：");
        string userName = Console.ReadLine();
        Console.Write("请输入密码：");
        string pwd = Console.ReadLine();
        Console.Write("请输入昵称：");
        string nick = Console.ReadLine();
        Console.Write("请输入Sex(0/1/2)：");
        int sex = int.Parse(Console.ReadLine());

        User newUser = new User
        {
            UserName = userName,
            PassWord = pwd,
            NickName = nick,
            Sex = sex
        };

        if (bll.AddUser(newUser))
        {
            Console.WriteLine("新增成功！");
        }
        else
        {
            Console.WriteLine("新增失败！");
        }
    }

    // 修改用户
    static void UpdateUserInfo(UserBLL bll)
    {
        Console.Write("请输入要修改的用户ID：");
        int id = int.Parse(Console.ReadLine());
        User user = bll.GetUserById(id);
        if (user == null)
        {
            Console.WriteLine("用户不存在！");
            return;
        }

        Console.Write($"新昵称（原昵称：{user.NickName}）：");
        string newNick = Console.ReadLine();
        Console.Write($"新Sex（原Sex：{user.Sex}）：");
        int newSex = int.Parse(Console.ReadLine());

        user.NickName = newNick;
        user.Sex = newSex;

        if (bll.UpdateUser(user))
        {
            Console.WriteLine("修改成功！");
        }
        else
        {
            Console.WriteLine("修改失败！");
        }
    }

    // 删除用户
    static void DeleteUserInfo(UserBLL bll)
    {
        Console.Write("请输入要删除的用户ID：");
        int id = int.Parse(Console.ReadLine());
        if (bll.DeleteUser(id))
        {
            Console.WriteLine("删除成功！");
        }
        else
        {
            Console.WriteLine("删除失败！");
        }
    }
}
```
