```c#
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data;
using System.Data.SqlClient;

namespace Common
{
    public class DBHelper
    {
        // 数据库连接字符串
        private static string connStr = "Data Source=DESKTOP-VL906FO\\SQLEXPRESS;Initial Catalog=shizhan;Integrated Security=True;Trust Server Certificate=True";


        //执行增删改
        public static int ExecuteNonQuery(string sql, SqlParameter[] arrP)  
        {
            using (SqlConnection con =new SqlConnection(connStr))  //using代码快执行完括号里的自动销毁
            {
                con.Open(); //打开数据库连接
                using (SqlCommand cmd = new SqlCommand(sql ,con))
                {
                    if(arrP != null) //执行的就是参数化的SQL指令
                    {
                        foreach (SqlParameter param in arrP) //遍历数组的每个元素
                        {
                            cmd.Parameters.Add(param);  //每个元素加入到cmd.Parameters里面
                        }
                    }
                    int i = cmd.ExecuteNonQuery();
                    return i;
                }
            }
            return 0;
        }
        
        //执行查询
        public static DataTable ExecuteQuery(string sql, SqlParameter[] arrP, bool isProc)  
        {
            using (SqlConnection con = new SqlConnection(connStr))  //using代码快执行完括号里的自动销毁
            {
                con.Open(); //打开数据库连接
                using (SqlCommand cmd = new SqlCommand(sql, con))
                {
                    cmd.CommandType = (isProc)?CommandType.StoredProcedure : CommandType.Text;
                    if (arrP != null) //执行的就是参数化的SQL指令
                    {
                        foreach (SqlParameter param in arrP) //遍历数组的每个元素
                        {
                            cmd.Parameters.Add(param);  //每个元素加入到cmd.Parameters里面
                        }
                    }
                    DataTable dt = new DataTable();  //用来保存查询结果的
                    //数据适配器
                    SqlDataAdapter da = new SqlDataAdapter(cmd);
                    da.Fill(dt); //将查询结果填充到dt表中
                    return dt;   //返回数据表
                }
            }
            
        }


    }
}
```

```c#
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Common;
using Model;
using System.Data;
using System.Data.SqlClient;
namespace DAL
{
    public class GoodsInfoDAL
    {
        public DataTable GoodsInfoList()  //查询商品信息
        {
            DataTable dt = new DataTable();
            string sql = "select * from t_goods";
            dt = DBHelper.ExecuteQuery(sql, null, false);
            return dt;
        }
        public goods SelectOne(int gid)  //根据商品编号获取一个完整的商品信息
        {
            goods g = new goods();
            string sql = "select * from t_goods where gid = @gid";
            SqlParameter[] arr =
            {
                new SqlParameter("@gid",gid)
            };
            DataTable dt = DBHelper.ExecuteQuery(sql,arr, false);
            g.GId = gid;
            g.GName = dt.Rows[0]["gname"].ToString();
            g.Price = Convert.ToInt32(dt.Rows[0]["price"]);
            g.Sale = Convert.ToInt32(dt.Rows[0]["sale"]);
            g.Type = dt.Rows[0]["type"].ToString();
            return g;
        }
    }
}
```

>最后是用户访问的UI界面层，这层需要创建对BLL层和Model层的依赖

```c#
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data;
using System.Data.SqlClient;
using Model;
using BLL;

namespace shopping
{
    class Program
    {
        static void menu(string userName)
        {
            Console.WriteLine("亲爱的{userName}，欢迎光临");
            Console.WriteLine("**************************");
            Console.WriteLine("         购物商场         ");
            Console.WriteLine("                          ");
            Console.WriteLine("  1.浏览商品              ");
            Console.WriteLine("  2.购物车                ");
            Console.WriteLine("  3.退出                  ");
            Console.WriteLine("**************************");
            Console.WriteLine("请输入（1-3）             ");
        }
        static void Main(string[] args)
        {
            Console.WriteLine("欢迎登陆：");
            Console.Write("账号：");
            string account = Console.ReadLine();
            Console.Write("密码：");
            string password = Console.ReadLine();

            UserInfoBLL userinfobll = new UserInfoBLL();
            GoodsInfoBLL goodsbll = new GoodsInfoBLL();
            ShopCarBLL shopcarbll = new ShopCarBLL();

            bool ok =userinfobll.Login(account, password);
            if (!ok)
            {
                Console.WriteLine("账号密码错误！");
            }
            else
            {
                //登陆成功后获取用户姓名
                Console.WriteLine("登陆成功");
                string userName = userinfobll.GetUserName(account);
                string mid; //菜单编号
                while (true)
                {
                    menu(userName);
                    mid = Console.ReadLine();
                    if(mid =="3") break;
                    else if(mid =="1") //显示商品
                    {
                        DataTable dt = goodsbll.GoodsInfoList();
                        foreach (DataRow row in dt.Rows)
                        {
                            Console.WriteLine("商品ID\t商品名\t商品类型\t商品价格\t商品销量");
                            Console.WriteLine($"{row["gid"]}\t{row["gname"]}\t\t{row["type"]}\t\t{row["price"]}\t\t{row["sale"]}");
                        }
                    }else if(mid =="2") //购物车
                    {
                        Console.WriteLine("请输入要购买的商品编号：");
                        int gid = Convert.ToInt32(Console.ReadLine());
                        goods g = goodsbll.SelectOne(gid);
                        Console.WriteLine("请输入要购买的数量：");
                        int amount = Convert.ToInt32(Console.ReadLine());
                        shopCar shopcar = new shopCar()
                        {
                            GId = gid,
                            GName = g.GName,
                            Price = g.Price,
                            Account = account , //当前登陆用户作为消费者
                            Money = g.Price * amount //计算本间商品的总金额 = 价格 * 数量
                        };
                        shopcarbll.Add(shopcar);
                    }
                }
                Console.WriteLine($"{userName}欢迎下次光临！");
            }

        }
    }
}
```
